    CMP0042 # CMake 3.0
    CMP0063 # CMake 3.3.2
    )

    cmake_policy(SET ${p} NEW)
  endif()
endforeach()

macro(add_language_subdirectory lang)
  string(TOUPPER ${lang} LANG)
  if ( WRAP_${LANG} )
    add_subdirectory(${lang})
  endif()
endmacro()
